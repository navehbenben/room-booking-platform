import React from 'react';
import { Link } from 'react-router-dom';
import { useCheckout } from '../hooks/useCheckout';
import { CountdownTimer } from '../components/checkout/CountdownTimer';
import { BookingSummary } from '../components/checkout/BookingSummary';
import { CheckoutForm } from '../components/checkout/CheckoutForm';
import { ExpiryModal } from '../components/checkout/ExpiryModal';
 
export function CheckoutPage() {
  const { hold, room, remainingSeconds, loading, error, confirm, expired, confirmLoading, bookingId } =
    useCheckout();
 
  // Non-blocking warning at <60s remaining
  const isWarning = remainingSeconds > 0 && remainingSeconds < 60;
 
  // ── Loading ──────────────────────────────────────────────────────────────
  if (loading) {
    return (
      <div className="main">
        <div className="checkout-page checkout-page--loading">Loading reservation…</div>
      </div>
    );
  }
 
  // ── Success ───────────────────────────────────────────────────────────────
  if (bookingId) {
    return (
      <div className="main">
        <div className="checkout-page checkout-page--success">
          <div className="checkout-success__icon">✓</div>
          <h2 className="checkout-success__title">Booking confirmed!</h2>
          <p className="checkout-success__id">Booking ID: {bookingId}</p>
          <Link to="/" className="btn btn--primary btn--md">
            Back to search
          </Link>
        </div>
      </div>
    );
  }
 
  // ── Expired before hold was ever loaded (e.g., stale URL) ─────────────────
  if (!hold && expired) {
    return (
      <div className="main">
        <div className="checkout-page checkout-page--expired">
          <h2>Reservation expired</h2>
          <p>This reservation has already expired. Please search again.</p>
          <Link to="/" className="btn btn--primary btn--md">
            Back to search
          </Link>
        </div>
      </div>
    );
  }
 
  // ── Not found / other error ───────────────────────────────────────────────
  if (error || !hold) {
    return (
      <div className="main">
        <div className="alert alert--error">{error ?? 'Reservation not found'}</div>
        <Link
          to="/"
          className="btn btn--ghost btn--md"
          style={{ marginTop: '1rem', display: 'inline-flex' }}
        >
          Back to search
        </Link>
      </div>
    );
  }
 
  // ── Checkout page (hold loaded; modal overlays if expired in-place) ────────
  return (
    <>
      {/* Blocking modal — shown when timer expires while user is on the page.
          Non-dismissable: only the "Check availability again" CTA closes it. */}
      {expired && (
        <ExpiryModal roomId={hold.roomId} start={hold.start} end={hold.end} />
      )}
 
      <div className="main">
        <div className="checkout-page">
          <CountdownTimer expiresAt={hold.expiresAt} />
 
          {/* Non-blocking warning banner at <60 s */}
          {isWarning && (
            <div className="checkout-warning-banner" role="alert">
              ⚠ Less than 60 seconds left — complete your booking now.
            </div>
          )}
 
          {error && <div className="alert alert--error">{error}</div>}
 
          <div className="checkout-layout">
            <div className="checkout-layout__summary">
              {room ? (
                <BookingSummary room={room} start={hold.start} end={hold.end} />
              ) : (
                <div className="checkout-layout__no-room">
                  <p>Start: {new Date(hold.start).toLocaleString()}</p>
                  <p>End: {new Date(hold.end).toLocaleString()}</p>
                </div>
              )}
            </div>
            <div className="checkout-layout__form">
              {/* disabled=expired prevents submit even if modal hasn't rendered yet */}
              <CheckoutForm onConfirm={confirm} loading={confirmLoading} disabled={expired} />
            </div>
          </div>
        </div>
      </div>
    </>
  );
}
